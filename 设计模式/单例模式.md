<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>单例模式</title>
  <style>
    * {
      margin: 0;
      padding: 0;
    }
    .alert {
      position: fixed;
      top: 50px;
      width: 100%;
      text-align: center;
    }
    .alert-text {
      display: inline-block;
      min-width: 300px;
      padding: 8px 15px;
      border-radius: 8px;
      box-shadow: -4px 4px 2px 0 #ddd;
      background-color: #409eff;
      color: #fff;
    }
  </style>
</head>
<body>
  <script>
    const Alert = (function () {
      function CreateAlert(msg) {
        this.dom = document.createElement('div');
        this.dom.classList.add('alert');
        this.dom.innerHTML = `<div class="alert-text">${msg}</div>`;
      }
      CreateAlert.prototype = {
        updateMsg(msg) {
          if(msg) {
            this.dom.querySelector('.alert-text').innerHTML = msg;
          }
        },
        show(msg) {
          this.updateMsg(msg);
          document.body.appendChild(this.dom);
        },
        hide() {
          document.body.removeChild(this.dom);
        }
      }
      let alert = null;
      return function (msg) {
        if(!alert) {
          alert = new CreateAlert(msg);
        }
        alert.updateMsg(msg);
        return alert;
      }
    })()

    const alert1 = new Alert('我是alert1');
    const alert2 = new Alert('我是alert2');
    console.log(alert1 === alert2); // true
  </script>
</body>
</html>
