    class Food {

        this.name = name;
      }

        return this.name;
      }
    }

    class HouseWork {
      constructor(name) {
        this.name = name;
      }
    
      getName() {
        return this.name;
      }
    }
    
    // 孩子对象
    const child = {
      buyFood(food) {
        mother.buyFood(food);
      },
      doHouseWork(houseWork) {
        mother.doHouseWork(houseWork);
      }
    }
    
    // 妈妈对象
    const mother = {

    // 图片对象
    const myImage = (function () {
      const image = new Image();
      document.body.appendChild(image);
      return {
        setSrc(src) {
          image.src = src;
        }
      }
    })()
    
    // 预加载对象
    const preImage = (function () {
      const image = new Image();
      image.onload = function () {
        myImage.setSrc(image.src);
      }
      return {
        setSrc(src) {
          myImage.setSrc('./loading.jpg');
          image.src = src;
        }
      }
    })()

  <script>
    /**
     * 缓存代理
     */
    function multAdd(...args) {
      let res = 0;
      for(let val of args) {
        res += val;
      }
      return res;
    }

    const proxyAdd = (function () {
      const cache = new Map();
      return function (...args) {
        const key = args.join(',');
        if(cache.has(key)) {
          return cache.get(key);
        }
        const val = multAdd(...args);
        cache.set(key, val);
        return val;
      }
    })()
    
    console.log(proxyAdd(1,2,3,4));
    console.log(proxyAdd(2,3,4,5));
  </script>
