          this.sex = sex;
          this.underware = underware;
        }
        takePhoto() {
          console.log(`sex=${this.sex}, underware=${this.underware}`);
        }
      }
      // 男士内衣
      for(let i = 1; i <= 50; i++) {
        const mensUnderware = new Underware('man', 'underware' + i);

        const womenUnderware = new Underware('woman', 'underware' + i);

      class MensUnderware {
        setUnderware(underware) {
          this.underware = underware;
        }
        takePhoto() {
          console.log(`sex=man, underware=${this.underware}`);
        }
      }
      class WomensUnderware {
        setUnderware(underware) {
          this.underware = underware;
        }
        takePhoto() {
          console.log(`sex=woman, underware=${this.underware}`);
        }
      }
    
      const manUnderware = new MensUnderware();
      const womanUnderware = new WomensUnderware();
    
      // 男士内衣
      for(let i = 1; i <= 50; i++) {
        manUnderware.setUnderware('underware' + i);
        manUnderware.takePhoto();
      }
